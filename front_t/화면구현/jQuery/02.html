객체 조작

속성조작

  • html() : $("선택자").html("

    안녕

    ");
  • text() : $("선택자").html("

    안녕

    ");
  • css() : $("선택자").css("color","red");
  • removeAttr() : $("선택자").removeAttr("title");
  • attr() : $("선택자").attr("title","안녕");
  • addClass() : $("선택자").addClass("red");
  • removeClass() : $("선택자").removeClass("red");
  • toggleClass() : $("선택자").removeClass("red");
  • hasClass() : $("선택자").hasClass("red") : 클래스값이 있으면 true, 없으면 false값 반환
  • prop : $("선택자").prop("속성명","값") : 클래스값이 있으면 true, 없으면 false값 반환
    - 속성명 : tagName / nodeType / selectIndex
  • val() : $("선택자").val("값")

수치조작

  • height(): $("선택자").height("100");
  • width(): $("선택자").width("100");
  • innerHeight() : $("선택자").innerHeight("100"); // border 안쪽(padding 포함)까지의 크기
  • innerWidth() : $("선택자").innerWidth("100");
  • outerHeight() : $("선택자").outerHeight("100");// border를 포함한 크기
  • outerWidth() : $("선택자").outerWidth("100");
  • position() : $("선택자").position().left;
    $("선택자").position().top;
    선택요소의 상대좌표값을 반환
  • offset() : $("선택요소").offset().left / top // 절대좌표값
  • scrollLeft() : $(window).scrollLeft() // 브라우저의 수평 스크롤 위치 값을 반환
  • scrollTop() : $(window).scrollTop() // 브라우저의 수평 스크롤 위치 값을 반환

객체편집

  • before() : $("선택자").before("새요소") : 선택자 이전위치에 추가
  • after() : $("선택자").after("새요소") : 선택자 다음위치에 추가
  • append() : $("선택자").append("새요소") : 선택자 자식으로 새요소를 마지막위치에 추가
  • appendTo() : $("새요소").appendTo("선택자") : 선택자 자식으로 새요소를 마지막위치에 추가
  • prepend() : $("선택자").prepend("새요소") : 선택자 자식으로 새요소를 처음 위치에 추가
  • prependTo() : $("새요소").prependTo("선택자") : 선택자 자식으로 새요소를 처음 위치에 추가
  • insertBefore() : $("새요소").insertBefore("선택자") : 선택자 이전 위치에 추가
  • insertAfter() : $("새요소").insertAfter("선택자") : 선택자 다음 위치에 추가
  • clone() : $("선택자").clone() : 콘텐츠를 포함한 소스자체를 복사, 사용시에는 변수에 받아 사용.
  • empty() : $("선택자").empty() : 선택자의 하위요소를 모두 삭제
  • remove() : $("선택자").remove() : 선택자를 삭제
  • replaceAll() : $("선택자").replaceAll("새요소") : 선택자를 새요소로 교체
  • replaceWith() : $("새요소").replaceWith("선택자") : 선택자를 새요소로 교체
  • wrap() : $("선택자").wrap("새요소") : 선택자를 새요소로 감쌈
  • unwrap() : $("선택자").wrap() : 선택자의 부모요소를 삭제
  • wrapAll() / wrapInner()
